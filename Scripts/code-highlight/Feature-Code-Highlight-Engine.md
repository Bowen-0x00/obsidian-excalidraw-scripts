const svgToBase64 = (svg) => `data:image/svg+xml;base64,${btoa(unescape(encodeURIComponent(svg.replaceAll("&nbsp;", " "))))}`;

function hasPropertiesChanged(a, b) {
  return Math.abs(a.width - b.width) > 0.5 || 
         Math.abs(a.height - b.height) > 0.5 || 
         a.text !== b.text || 
         JSON.stringify(a?.customData?.codeHighlight) !== JSON.stringify(b?.customData?.codeHighlight);
}

let _codeBlockUpdateTimer = null;
const forceUpdateAllCodeBlocks = () => {
  if (_codeBlockUpdateTimer) clearTimeout(_codeBlockUpdateTimer);
  
  _codeBlockUpdateTimer = setTimeout(() => {
    if (!ea) return;
    const api = ea.getExcalidrawAPI();
    if (!api) return;

    const elements = api.getSceneElements();
    let hasChanges = false;
    
    const nextElements = elements.map(el => {
      if (el?.customData?.codeHighlight) {
        hasChanges = true;
        return { 
          ...el, 
          version: (el.version || 0) + 1, 
          versionNonce: Math.floor(Math.random() * 1000000000) 
        };
      }
      return el;
    });

    if (hasChanges) {
      console.log(t("log_batched_update"));
      api.updateScene({ elements: nextElements });
    }
  }, 100);
};

function loadHljs() {
  if (window._hljs_loaded) return;

  if (!document.getElementById("hljs")) {
    let script = document.createElement("script");
    script.id = "hljs";
    script.textContent = hljsContent; 
    document.head.appendChild(script);
  } 

  if (!document.getElementById("hljs-verilog")) {
    let scriptVerilog = document.createElement("script");
    scriptVerilog.id = "hljs-verilog";
    scriptVerilog.textContent = verilog;
    document.head.appendChild(scriptVerilog);
  }
  
  if (!document.getElementById("hljs-gdscript")) {
    let scriptGd = document.createElement("script");
    scriptGd.id = "hljs-gdscript";
    scriptGd.textContent = gdscript;
    document.head.appendChild(scriptGd);
  }
  
  window._hljs_loaded = true;
  forceUpdateAllCodeBlocks();
}

const handleDraw = (hookPayload) => {
    const { ea, api } = hookPayload;
    if (!ea || !api) return;

    const { element, context, renderConfig, rc } = hookPayload;
    if (!element?.customData?.codeHighlight) return false;

    if (typeof hljs === "undefined") {
      loadHljs();
      return false; 
  }

    const codeMap = getCodeMap();
    const cache = codeMap.get(element.id);
    let image;

    if (!cache || hasPropertiesChanged(cache.element, element)) {
      const lines = element.text.split("\n");
      const codes = [];
      const language = element.customData.codeHighlight.language;
      lines.forEach(line => {
        codes.push(hljs.highlight(line, {language: language}));
      });

      var div = document.createElement('div');
      div.className = `hljs language-${language}`;
      div.innerHTML = codes.map(code => code.value).join("\n") + "\n";
      div.style.width = `${element.width+20}px`;
      div.style.height = `${element.height+20}px`;
      div.style.fontSize = `${element.fontSize}px`;
      div.style.whiteSpace = "pre-wrap";
      div.style.padding = "10px 10px 10px 10px";
      div.style.fontFamily = ExcalidrawLib.getFontFamilyString({fontFamily:element.fontFamily});
      div.style.lineHeight = element.lineHeight;
      div.style.boxSizing = "content-box";
      div.style.letterSpacing = "initial";
      div.style.textWrap = "nowrap";
      
      const xml = new XMLSerializer().serializeToString(div);
      let fontDef;
      if (element?.customData?.codeHighlight?.embedFont) {
        fontDef = ExcalidrawLib.getCSSFontDefinition2(element.fontFamily);
      }
      
      const svg = `<svg xmlns="http://www.w3.org/2000/svg" width="${element.width+20}" height="${element.height+20}">
      <style>${codeHighlightStyle.style[element?.customData?.codeHighlight?.style]}</style>
      <foreignObject x="0" y="0" width="100%" height="100%">
      ${xml}
      </foreignObject>
      ${fontDef ? `<defs><style>${fontDef}</style></defs>` : ""}
      </svg>`;
      
      const base64 = svgToBase64(svg);
      image = new Image();
      image.onload = () => {
        forceUpdateAllCodeBlocks();
      };
      image.src = base64;
      
      codeMap.set(element.id, { element, svg, image });
      
      if (!image.complete) {
          return false;
      }
    } else {
        image = cache.image;
    }

    if (!image || !image.complete) return false;

    context.drawImage(image, -10, -10, element.width + 20, element.height + 20);
    return true; 
};

const handleRenderCheck = (context) => !!context.element?.customData?.codeHighlight;

const handleRenderSvg = (context) => {
    const { ea, api } = context;
    if (!ea || !api) return;

    const { addToRoot, element, offsetX, offsetY, degree, cx, cy } = context;
    if (!element?.customData?.codeHighlight) return;
    
    const codeMap = getCodeMap();
    const cache = codeMap.get(element.id); 
    if (!cache?.svg) return;

    const dom = new DOMParser().parseFromString(cache.svg, 'image/svg+xml');
    const svgNode = dom.documentElement; // image/svg+xml 解析出的直接就是 svg 根节点
    if (!svgNode || svgNode.nodeName === "parsererror") return;

    const gElement = document.createElementNS('http://www.w3.org/2000/svg', 'g');
    Array.from(svgNode.attributes).forEach(attr => gElement.setAttribute(attr.name, attr.value));
    while (svgNode.firstChild) gElement.appendChild(svgNode.firstChild);

    gElement.setAttribute("transform", `translate(${(offsetX || 0) - 20} ${(offsetY || 0) - 20}) rotate(${degree} ${cx} ${cy})`);
    gElement.classList.add(element.customData.codeHighlight.style);
    
    const styleDef = gElement.querySelector("style");
    if (styleDef) styleDef.innerHTML = styleDef.innerHTML.replace(/\}\./g, `} .${element.customData.codeHighlight.style} .`);
    
    addToRoot(gElement, element);
};

const handlePaste = async (contextPayload) => {
    const { ea, api } = contextPayload;
    if (!ea || !api) return;

    const { ea: currentEa, payload } = contextPayload;
    if (!payload?.text?.startsWith("ymjr:")) return true;
    
    try {
        const obj = JSON.parse(payload.text.substring(5));
        if (obj?.type === 'code') {
            payload.text = '';
            currentEa.clear();
            const id = await currentEa.addText(0, 0, obj.text.replaceAll('\r\n', '\n'));
            const el = currentEa.getElement(id);
            el.fontFamily = 3;
            el.link = `ymjr://open?app=code&folder=${encodeURI(obj.folderPath)}&file=${encodeURI(obj.filePath)}&line=${obj.lineNum}`;
            el.customData = { codeHighlight: { embedFont: true, language: obj.language || "Verilog", style: "atom-one-dark" } };
            currentEa.refreshTextElementSize(id);
            await currentEa.addElementsToView(true);
            return false;
        }
    } catch(e) {}
    return true; 
};

async function mountFeature() {
    if (!window.EA_Core) {
        console.warn(t("log_mount_failed", { id: SCRIPT_ID }));
        return;
    }
    
    if (typeof ExcalidrawAutomate.plugin[`disable_${SCRIPT_ID}`] === "function") {
      ExcalidrawAutomate.plugin[`disable_${SCRIPT_ID}`]();
    }

    window.EA_Core.registerHook(SCRIPT_ID, 'drawElementOnCanvas', handleDraw, 60);
    window.EA_Core.registerHook(SCRIPT_ID, 'renderElementToSvgCheck', handleRenderCheck, 60);
    window.EA_Core.registerHook(SCRIPT_ID, 'renderElementToSvg', handleRenderSvg, 60);
    window.EA_Core.registerHook(SCRIPT_ID, 'onPaste', handlePaste, 60);

    ExcalidrawAutomate.plugin[`disable_${SCRIPT_ID}`] = () => {
        if (window.EA_Core) {
            window.EA_Core.unregisterHook(SCRIPT_ID);
        }
        console.log(t("log_unmounted", { id: SCRIPT_ID }));
    };

    console.log(t("log_mounted", { id: SCRIPT_ID }));
}
